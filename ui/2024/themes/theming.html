// General

@primary-color: #cfe5fa;     

// Base color for bar-backgrounds
@secondary-color: #efefef;    
// Text color for enabled widgets
@text-color: #000;    
// Base for disabled backgrounds and borders
@disabled-color: #d3d3d3;    
@error-color: #d46464;    
// Backgrounds for various content areas such as TitlePane, ContentPane and Inputs (if changed, adjust selected tab to match)
@container-background-color:#fff;    
// Color for various arrows and buttons
@minor-selected-color: spin(saturate(darken(@primary-color, 6), 19), 0);
// Augmented and used directly by variables to create border colors for various widgets
@base-border-color: spin(desaturate(darken(@primary-color, 29), 44), -1);
// Background color for enabled buttons, text inputs
@unfocused-clickable-color: spin(saturate(lighten(@primary-color, 5), 10), 0);
// Border color for (enabled, unhovered) TextBox, Slider, Accordion, BorderContainer, TabContainer
@border-color: spin(desaturate(darken(@primary-color, 15), 67), 8);
// Color of borders inside widgets: horizontal line in Calendar between weeks, around color swatches in ColorPalette, above Dialog action bar
@minor-border-color: @disabled-color;
// Border for Dialog, Menu, Tooltip. Must also update tooltip.png (the arrow image file) to match
@popup-border-color: @base-border-color;
// Border color for disabled/readonly Button, TextBox etc. widgets
@disabled-border-color: @disabled-color;
// Disabled button, textbox, etc.
@disabled-background-color: @secondary-color;
// Text color for disabled/readonly widgets
@disabled-text-color: darken(@secondary-color, 43);
// Background color for unselected/unopened tab button, accordion pane, TitlePane, Menu items
@unselected-background-color: @secondary-color;
// Text color for unselected/unopened tab button, accordion pane, TitlePane, Menu items
@unselected-text-color: darken(@secondary-color, 65);
// Hover of textbox, tab label, BorderContainer splitter, Calendar, etc.
@hovered-border-color: @base-border-color;
// Background color for hover of Button, MenuBar, Accordion pane, Calendar... anything that has a (non-white) color to start with and gets darker on hover
@hovered-background-color: @minor-selected-color;
// Used for title of select Accordion pane, label of select tab, hovered Menu item, etc.
@hovered-text-color: @text-color;
// During click on Calendar day, Slider up/down buttons, tab button, etc.
@pressed-border-color: @base-border-color;
// Background color while clicking on Accordion/TitlePane title bar, tab button, Calendar day, Toolbar button, Tree row.
@pressed-background-color: spin(saturate(darken(@primary-color, 16), 12), 0);
// Selected AccordionPane, tab of nested TabContainer (but plain TabContainer is special)
@selected-border-color: @base-border-color; // Selected Accordion pane, nested tab label, Tree row
@selected-background-color: @primary-color;
// title of selected Accordion pane, label of selected tab, hovered Menu item, etc.
@selected-text-color: @text-color;
// MenuBar, Toolbar, action bar at bottom of dialog
@bar-background-color: @secondary-color;
